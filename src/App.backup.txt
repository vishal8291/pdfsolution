  const [dragActive, setDragActive] = useState(false);

  useEffect(() => {
    let cancelled = false;

    async function loadPreviews() {
      const pdfFiles = files.filter((item) => isPdfFile(item.file) && !item.error);
      if (pdfFiles.length === 0) {
        setPreviewMap({});
        setMergeComposerPages([]);
        return;
      }

      setPreviewLoading(true);
      try {
        const entries = await Promise.all(
          pdfFiles.map(async (item) => {
            const preview = await buildPdfPreview(item.file);
            return [item.id, preview] as const;
          }),
        );

        if (cancelled) {
          return;
        }

        const nextPreviewMap = Object.fromEntries(entries);
        setPreviewMap(nextPreviewMap);

        if (activeTool === "split") {
          const firstPreview = pdfFiles[0] ? nextPreviewMap[pdfFiles[0].id] : null;
          if (firstPreview) {
            const firstPage = firstPreview.pageCount > 0 ? [1] : [];
            setVisualSplitPages(firstPage);
            setSplitSelection(firstPage.join(","));
          }
        }

        if (activeTool === "merge") {
          const nextMergePages = pdfFiles.flatMap((item) => {
            const preview = nextPreviewMap[item.id];
            if (!preview) {
              return [];
            }

            return preview.thumbnails.map((thumbnail, index) => ({
              id: `${item.id}-${thumbnail.pageNumber}`,
              fileId: item.id,
              fileName: item.file.name,
              file: item.file,
              pageIndex: index,
              pageNumber: thumbnail.pageNumber,
              image: thumbnail.image,
            }));
          });

          setMergeComposerPages(nextMergePages);
        } else {
          setMergeComposerPages([]);
        }
      } catch (previewError) {
        if (!cancelled) {
          const message = previewError instanceof Error ? previewError.message : "Unable to render preview.";
          setError(message);
          setPreviewMap({});
          setMergeComposerPages([]);
        }
      } finally {
        if (!cancelled) {
          setPreviewLoading(false);
        }
      }
    }

    void loadPreviews();

    return () => {
      cancelled = true;
    };
  }, [files, activeTool]);

  async function handleFileSelection(event: ChangeEvent<HTMLInputElement>) {
    const nextFiles = Array.from(event.target.files ?? []);
    await addFiles(nextFiles);
    event.target.value = "";
  }

  async function addFiles(incoming: File[]) {
    if (incoming.length === 0) {
      return;
    }

    setError("");
    setStatus(`Loading ${incoming.length} file(s)...`);

    const mapped = await Promise.all(
      incoming.map(async (file, index) => {
        const id = `${file.name}-${file.size}-${index}-${Date.now()}`;
        if (isPdfFile(file)) {
          try {
            const pageCount = await getPdfPageCount(file);
            return { id, file, pageCount } satisfies UploadedFile;
          } catch (loadError) {
            const message = loadError instanceof Error ? loadError.message : "Unable to read PDF.";
            return { id, file, error: message } satisfies UploadedFile;
          }
        }

        return { id, file } satisfies UploadedFile;
      }),
    );

    setFiles((current) => {
      const next = tool.multi ? [...current, ...mapped] : mapped.slice(0, 1);
      setStatus(`${next.length} file(s) ready.`);
      return next;
    });
  }

  async function runTool() {
    setBusy(true);
    setError("");
    setStatus(
      enableOcr && (activeTool === "extract" || activeTool === "word")
        ? `Running ${tool.title} with OCR...`
        : `Running ${tool.title}...`,
    );

    try {
      const output = await executeTool();
      const nextResults = output.map((item, index) => ({
        ...item,
        id: `${item.name}-${index}-${Date.now()}`,
        url: URL.createObjectURL(item.blob),
      }));

      setResults((current) => {
        current.forEach((result) => URL.revokeObjectURL(result.url));
        return nextResults;
      });
      setStatus(`${tool.title} finished. ${nextResults.length} file(s) ready to download.`);
    } catch (runError) {
      const message = runError instanceof Error ? runError.message : "Something went wrong.";
      setError(message);
      setStatus("Action failed.");
    } finally {
      setBusy(false);
    }
  }

  async function downloadAllResults() {
    setBusy(true);
    setError("");

    try {
      const zipResult = await bundleResultsAsZip(
        results.map(({ name, blob, summary, textPreview }) => ({ name, blob, summary, textPreview })),
        `${tool.id}-results.zip`,
      );
      const url = URL.createObjectURL(zipResult.blob);
      triggerDownload(url, zipResult.name);
      setTimeout(() => URL.revokeObjectURL(url), 1000);
      setStatus("Bundled all result files into a ZIP download.");
    } catch (bundleError) {
      const message = bundleError instanceof Error ? bundleError.message : "Unable to create ZIP.";
      setError(message);
    } finally {
      setBusy(false);
    }
  }

  async function executeTool(): Promise<ResultFile[]> {
    ensureValidSelection();

    switch (activeTool) {
      case "merge": {
        const mergePlan: MergePlanItem[] = mergeComposerPages.map((page) => ({
          file: page.file,
          pageIndex: page.pageIndex,
        }));
        return [await mergePdfFiles(files.map((item) => item.file), mergePlan)];
      }
      case "split":
        return splitPdfFile(files[0].file, splitSelection, splitEveryPage);
      case "extract":
        return [await extractTextFromPdf(files[0].file, { useOcr: enableOcr })];
      case "edit":
        return [
          await editPdfFile(files[0].file, {
            rotation: Number(rotation) || 0,
            removePages,
            watermark,
          }),
        ];
      case "word":
        return [await convertPdfToWord(files[0].file, { useOcr: enableOcr })];
      case "imageToPdf":
        return [await imagesToPdf(files.map((item) => item.file))];
      case "compress":
        return [await compressPdfFile(files[0].file)];
      default:
        throw new Error("Unsupported tool.");
    }
  }

  function ensureValidSelection() {
    if (files.length === 0) {
      throw new Error("Upload a file before running a tool.");
    }

    if (files.some((item) => item.error)) {
      throw new Error("One or more files could not be read. Replace those files and try again.");
    }

    if (activeTool === "merge" && files.length < 2) {
      throw new Error("Merge needs at least two PDF files.");
    }

    if (activeTool === "merge" && mergeComposerPages.length === 0) {
      throw new Error("No merge pages are available yet.");
    }

    if (activeTool === "imageToPdf" && files.some((item) => isPdfFile(item.file))) {
      throw new Error("Image to PDF only accepts JPG and PNG files.");
    }
  }

  function removeFile(id: string) {
    setFiles((current) => current.filter((item) => item.id !== id));
  }

  function moveFile(id: string, direction: -1 | 1) {
    setFiles((current) => {
      const index = current.findIndex((item) => item.id === id);
      const nextIndex = index + direction;
      if (index < 0 || nextIndex < 0 || nextIndex >= current.length) {
        return current;
      }

      const clone = [...current];
      const [file] = clone.splice(index, 1);
      clone.splice(nextIndex, 0, file);
      return clone;
    });
  }

  function moveMergePage(id: string, direction: -1 | 1) {
    setMergeComposerPages((current) => {
      const index = current.findIndex((item) => item.id === id);
      const nextIndex = index + direction;
      if (index < 0 || nextIndex < 0 || nextIndex >= current.length) {
        return current;
      }

      const clone = [...current];
      const [page] = clone.splice(index, 1);
      clone.splice(nextIndex, 0, page);
      return clone;
    });
  }

  function handleDropAreaDragOver(event: DragEvent<HTMLLabelElement>) {
    event.preventDefault();
    setDragActive(true);
  }

  function handleDropAreaDragLeave(event: DragEvent<HTMLLabelElement>) {
    event.preventDefault();
    setDragActive(false);
  }

  async function handleDropAreaDrop(event: DragEvent<HTMLLabelElement>) {
    event.preventDefault();
    setDragActive(false);
    const droppedFiles = Array.from(event.dataTransfer.files);
    await addFiles(droppedFiles);
  }

  function toggleSplitPage(pageNumber: number) {
    if (splitEveryPage) {
      return;
    }

    setVisualSplitPages((current) => {
      const exists = current.includes(pageNumber);
      const next = exists ? current.filter((page) => page !== pageNumber) : [...current, pageNumber];
      const sorted = [...next].sort((left, right) => left - right);
      setSplitSelection(sorted.join(","));
      return sorted;
    });
  }

  function handleSplitSelectionChange(value: string) {
    setSplitSelection(value);
    const parsed = value
      .split(",")
      .map((segment) => Number(segment.trim()))
      .filter((page) => Number.isInteger(page) && page > 0);
    setVisualSplitPages([...new Set(parsed)].sort((left, right) => left - right));
  }

  function handleContactFieldChange(field: keyof ContactForm, value: string) {
    setContactForm((current) => ({
      ...current,
      [field]: value,
    }));
  }

  async function handleContactSubmit(event: FormEvent<HTMLFormElement>) {
    event.preventDefault();
    setContactStatus("");
    setContactSubmitting(true);

    try {
      const response = await fetch("/api/contact", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          name: contactForm.name.trim(),
          email: contactForm.email.trim(),
          message: contactForm.message.trim(),
        }),
      });

      const result = (await response.json()) as { message?: string };
      if (!response.ok) {
        throw new Error(result.message ?? "Unable to send your message right now.");
      }

      setContactStatus(result.message ?? "Thanks, your message has been saved.");
      setContactForm({
        name: "",
        email: "",
        message: "",
      });
    } catch (contactRequestError) {
      const isNetworkIssue = contactRequestError instanceof TypeError;
      const message = isNetworkIssue
        ? "Contact server is not available right now. Start `npm.cmd run dev:server` and try again."
        : contactRequestError instanceof Error
          ? contactRequestError.message
          : "Unable to send your message right now.";
      setContactStatus(message);
    } finally {
      setContactSubmitting(false);
    }
  }

    setMobileNavOpen(false);

  function handleAuthFieldChange(field: keyof AuthForm, value: string) {
    setAuthForm((current) => ({
      ...current,
      [field]: value,
    }));
  }

  async function handleAuthSubmit(event: FormEvent<HTMLFormElement>) {
    event.preventDefault();
    setAuthError("");

    const email = authForm.email.trim().toLowerCase();
    const password = authForm.password.trim();
    const name = authForm.name.trim();

    if (!email || !password || (authMode === "signup" && !name)) {
      setAuthError("Fill in all required fields to continue.");
      return;
    }

    if (password.length < 6) {
      setAuthError("Password must be at least 6 characters long.");
      return;
    }

    const endpoint = authMode === "signup" ? "/api/auth/signup" : "/api/auth/login";
    const payload = authMode === "signup" ? { name, email, password } : { email, password };

    try {
      const response = await fetch(endpoint, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify(payload),
      });

      const result = (await response.json()) as { message?: string; token?: string; user?: SessionUser };
      if (!response.ok || !result.token || !result.user) {
        throw new Error(result.message ?? "Unable to authenticate right now.");
      }

      window.localStorage.setItem(authTokenStorageKey, result.token);
      setSessionUser(result.user);
      setAuthOpen(false);
      setAuthForm({
        name: "",
        email: "",
        password: "",
      });
    } catch (authRequestError) {
      const isNetworkIssue = authRequestError instanceof TypeError;
      const message = isNetworkIssue
        ? "Auth server is not running. Start `npm.cmd run dev:server` and try again."
        : authRequestError instanceof Error
          ? authRequestError.message
          : "Unable to authenticate right now.";
      setAuthError(message);
    }
  }

  async function handleLogout() {
    const token = window.localStorage.getItem(authTokenStorageKey);
    if (token) {
      await fetch("/api/auth/logout", {
        method: "POST",
        headers: {
          Authorization: `Bearer ${token}`,
        },
      }).catch(() => undefined);
    }

    window.localStorage.removeItem(authTokenStorageKey);
    setSessionUser(null);
    setAuthMode("login");
    setAuthError("");
    setAuthForm({
      name: "",
      email: "",
      password: "",
    });
  }

function isPdfFile(file: File) {
  return file.type === "application/pdf" || file.name.toLowerCase().endsWith(".pdf");
}

function formatBytes(value: number) {
  if (value < 1024) {
    return `${value} B`;
  }

  if (value < 1024 * 1024) {
    return `${(value / 1024).toFixed(1)} KB`;
  }

  return `${(value / (1024 * 1024)).toFixed(2)} MB`;
}

function triggerDownload(url: string, fileName: string) {
  const link = document.createElement("a");
  link.href = url;
  link.download = fileName;
  document.body.appendChild(link);
  link.click();
  link.remove();
}
